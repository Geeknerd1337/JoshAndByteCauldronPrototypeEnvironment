537  void AutoSetAnchorControlPoints (int anchorIndex) {
538  // Calculate a vector that is perpendicular to the vector bisecting the angle between this anchor and its two immediate neighbours
539  // The control points will be placed along that vector
540  Vector3 anchorPos = points[anchorIndex];
541  Vector3 dir = Vector3.zero;
542  float[] neighbourDistances = new float[2];
543 
544  if (anchorIndex - 3 >= 0 || isClosed) {
545  Vector3 offset = points[LoopIndex (anchorIndex - 3)] - anchorPos;
546  dir += offset.normalized;
547  neighbourDistances[0] = offset.magnitude;
548  }
549  if (anchorIndex + 3 >= 0 || isClosed) {
550  Vector3 offset = points[LoopIndex (anchorIndex + 3)] - anchorPos;
551  dir -= offset.normalized;
552  neighbourDistances[1] = -offset.magnitude;
553  }
554 
555  dir.Normalize ();
556 
557  // Set the control points along the calculated direction, with a distance proportional to the distance to the neighbouring control point
558  for (int i = 0; i < 2; i++) {
559  int controlIndex = anchorIndex + i * 2 - 1;
560  if (controlIndex >= 0 && controlIndex < points.Count || isClosed) {
561  points[LoopIndex (controlIndex)] = anchorPos + dir * neighbourDistances[i] * autoControlLength;
562  }
563  }

567  void AutoSetStartAndEndControls () {
568  if (isClosed) {
569  // Handle case with only 2 anchor points separately, as will otherwise result in straight line ()
570  if (NumAnchorPoints == 2) {
571  Vector3 dirAnchorAToB = (points[3] - points[0]).normalized;
572  float dstBetweenAnchors = (points[0] - points[3]).magnitude;
573  Vector3 perp = Vector3.Cross (dirAnchorAToB, (space == PathSpace.xy) ? Vector3.forward : Vector3.up);
574  points[1] = points[0] + perp * dstBetweenAnchors / 2f;
575  points[5] = points[0] - perp * dstBetweenAnchors / 2f;
576  points[2] = points[3] + perp * dstBetweenAnchors / 2f;
577  points[4] = points[3] - perp * dstBetweenAnchors / 2f;
578 
579  } else {

581  AutoSetAnchorControlPoints (points.Count - 3);

599  // The axis with the smallest bounds will be discarded.
600  if (previousSpace == PathSpace.xyz) {

603

605  if (space == PathSpace.xy) {
606  float x = (minBoundsSize == boundsSize.x) ? points[i].z : points[i].x;
607  float y = (minBoundsSize == boundsSize.y) ? points[i].z : points[i].y;
608  points[i] = new Vector3 (x, y, 0);
609  } else if (space == PathSpace.xz) {

611  float z = (minBoundsSize == boundsSize.z) ? points[i].y : points[i].z;

614  }
