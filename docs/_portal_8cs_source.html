138  // 1. Tiny sliver of mesh drawn on backside of portal
139  // Ideally the oblique clip plane would sort this out, but even with 0 offset, tiny sliver still visible
140  // 2. Tiny seam between the sliced mesh, and the rest of the model drawn onto the portal screen

142  // Would be great if this could be fixed more elegantly, but this is the best I can figure out for now
143  const float hideDst = -1000;
144  const float showDst = 1000;
145  float screenThickness = linkedPortal.ProtectScreenFromClipping (portalCam.transform.position);

262  // Use custom projection matrix to align portal camera's near clip plane with the surface of the portal
263  // Note that this affects precision of the depth buffer, which can cause issues with effects like screenspace AO
264  void SetNearClipPlane () {
265  // Learning resource:
266  // http://www.terathon.com/lengyel/Lengyel-Oblique.pdf
267  Transform clipPlane = transform;
268  int dot = System.Math.Sign (Vector3.Dot (clipPlane.forward, transform.position - portalCam.transform.position));
269 
270  Vector3 camSpacePos = portalCam.worldToCameraMatrix.MultiplyPoint (clipPlane.position);
271  Vector3 camSpaceNormal = portalCam.worldToCameraMatrix.MultiplyVector (clipPlane.forward) * dot;
272  float camSpaceDst = -Vector3.Dot (camSpacePos, camSpaceNormal) + nearClipOffset;
273 
274  // Don't use oblique clip plane if very close to portal as it seems this can cause some visual artifacts
275  if (Mathf.Abs (camSpaceDst) > nearClipLimit) {
276  Vector4 clipPlaneCameraSpace = new Vector4 (camSpaceNormal.x, camSpaceNormal.y, camSpaceNormal.z, camSpaceDst);
277 
278  // Update projection based on new clip plane
279  // Calculate matrix with player cam so that player camera settings (fov, etc) are used
280  portalCam.projectionMatrix = playerCam.CalculateObliqueMatrix (clipPlaneCameraSpace);
281  } else {
282  portalCam.projectionMatrix = playerCam.projectionMatrix;
283  }
