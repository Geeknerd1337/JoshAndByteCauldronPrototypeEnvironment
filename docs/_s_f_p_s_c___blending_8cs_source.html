510

513  matched |= ApplyIfMatch(BlendModePreset.Additive, BlendMode.One, BlendMode.One );
514  matched |= ApplyIfMatch(BlendModePreset.Screen, BlendMode.OneMinusDstColor, BlendMode.One );
515  matched |= ApplyIfMatch(BlendModePreset.Screen, BlendMode.One, BlendMode.OneMinusSrcColor );
516  matched |= ApplyIfMatch(BlendModePreset.AlphaBlended, BlendMode.SrcAlpha, BlendMode.OneMinusSrcAlpha );
517  matched |= ApplyIfMatch(BlendModePreset.AlphaBlendedPremultiplied, BlendMode.One, BlendMode.OneMinusSrcAlpha );
518  matched |= ApplyIfMatch(BlendModePreset.Multiplicative, BlendMode.DstColor, BlendMode.Zero );
519  matched |= ApplyIfMatch(BlendModePreset.Multiplicative, BlendMode.Zero, BlendMode.SrcColor );
520 
521  if(!matched){
522  blendModePreset = BlendModePreset.Custom;
523  }
