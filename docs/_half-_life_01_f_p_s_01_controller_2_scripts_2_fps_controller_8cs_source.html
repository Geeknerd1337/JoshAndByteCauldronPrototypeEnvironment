243  {

245  var projSpeed = Vector3.Dot(playerVelocity, accelDir);
246 
247  // How much speed we need to add (in that direction) to reach max speed

316  // Calculates the displacement required in order not to be in a world collider
317  private Vector3 ResolveCollisions(ref Vector3 playerVelocity)
318  {
319  // Get nearby colliders
320  Physics.OverlapSphereNonAlloc(_transform.position, Radius + 0.1f,
321  _overlappingColliders, ~_excludedLayers);
322 
323  var totalDisplacement = Vector3.zero;
324  var checkedColliderIndices = new HashSet<int>();
325 
326  // If the player is intersecting with that environment collider, separate them
327  for (var i = 0; i < _overlappingColliders.Length; i++)
328  {
329  // Two player colliders shouldn't resolve collision with the same environment collider
330  if (checkedColliderIndices.Contains(i))
331  {
332  continue;
333  }
334 
335  var envColl = _overlappingColliders[i];
336 
337  // Skip empty slots
338  if (envColl == null)
339  {
340  continue;
341  }
342 
343  Vector3 collisionNormal;
344  float collisionDistance;
345  if (Physics.ComputePenetration(
346  _collisionVolume, _collisionVolume.transform.position, _collisionVolume.transform.rotation,
347  envColl, envColl.transform.position, envColl.transform.rotation,
348  out collisionNormal, out collisionDistance))
349  {
350  // Ignore very small penetrations
351  // Required for standing still on slopes
352  // ... still far from perfect though
353  if (collisionDistance < 0.015)
354  {
355  continue;
356  }
357 
358  checkedColliderIndices.Add(i);
359 
360  // Get outta that collider!
361  totalDisplacement += collisionNormal * collisionDistance;
362 
363  // Crop down the velocity component which is in the direction of penetration
364  playerVelocity -= Vector3.Project(playerVelocity, collisionNormal);
365  }
366  }
367 
368  // It's better to be in a clean state in the next resolve call
369  for (var i = 0; i < _overlappingColliders.Length; i++)
370  {
371  _overlappingColliders[i] = null;
372  }
373 
374  return totalDisplacement;
375  }
