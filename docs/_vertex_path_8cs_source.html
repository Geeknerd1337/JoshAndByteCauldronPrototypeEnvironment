60  this.transform = transform;

71  bounds = new Bounds ((pathSplitData.minMax.Min + pathSplitData.minMax.Max) / 2, pathSplitData.minMax.Max - pathSplitData.minMax.Min);
72 
73  // Figure out up direction for path
74  up = (bounds.size.z > bounds.size.y) ? Vector3.up : -Vector3.forward;
75  Vector3 lastRotationAxis = up;
76 
77  // Loop through the data and assign to arrays.
78  for (int i = 0; i < localPoints.Length; i++) {
79  localPoints[i] = pathSplitData.vertices[i];
80  localTangents[i] = pathSplitData.tangents[i];
81  cumulativeLengthAtEachVertex[i] = pathSplitData.cumulativeLength[i];
82  times[i] = cumulativeLengthAtEachVertex[i] / length;
83 
84  // Calculate normals
85  if (space == PathSpace.xyz) {
86  if (i == 0) {
87  localNormals[0] = Vector3.Cross (lastRotationAxis, pathSplitData.tangents[0]).normalized;
88  } else {
89  // First reflection
90  Vector3 offset = (localPoints[i] - localPoints[i - 1]);
91  float sqrDst = offset.sqrMagnitude;
92  Vector3 r = lastRotationAxis - offset * 2 / sqrDst * Vector3.Dot (offset, lastRotationAxis);
93  Vector3 t = localTangents[i - 1] - offset * 2 / sqrDst * Vector3.Dot (offset, localTangents[i - 1]);
94 
95  // Second reflection
96  Vector3 v2 = localTangents[i] - t;
97  float c2 = Vector3.Dot (v2, v2);
98 
99  Vector3 finalRot = r - v2 * 2 / c2 * Vector3.Dot (v2, r);
100  Vector3 n = Vector3.Cross (finalRot, localTangents[i]).normalized;
101  localNormals[i] = n;
102  lastRotationAxis = finalRot;
103  }
104  } else {
105  localNormals[i] = Vector3.Cross (localTangents[i], up) * ((bezierPath.FlipNormals) ? 1 : -1);
106  }
107  }
108 
109  // Apply correction for 3d normals along a closed path
110  if (space == PathSpace.xyz && isClosedLoop) {
111  // Get angle between first and last normal (if zero, they're already lined up, otherwise we need to correct)
112  float normalsAngleErrorAcrossJoin = Vector3.SignedAngle (localNormals[localNormals.Length - 1], localNormals[0], localTangents[0]);
113  // Gradually rotate the normals along the path to ensure start and end normals line up correctly
114  if (Mathf.Abs (normalsAngleErrorAcrossJoin) > 0.1f) // don't bother correcting if very nearly correct
115  {
116  for (int i = 1; i < localNormals.Length; i++) {
117  float t = (i / (localNormals.Length - 1f));
118  float angle = normalsAngleErrorAcrossJoin * t;
119  Quaternion rot = Quaternion.AngleAxis (angle, localTangents[i]);
120  localNormals[i] = rot * localNormals[i] * ((bezierPath.FlipNormals) ? -1 : 1);
121  }
122  }
123  }
124 
125  // Rotate normals to match up with user-defined anchor angles
126  if (space == PathSpace.xyz) {
127  for (int anchorIndex = 0; anchorIndex < pathSplitData.anchorVertexMap.Count - 1; anchorIndex++) {
128  int nextAnchorIndex = (isClosedLoop) ? (anchorIndex + 1) % bezierPath.NumSegments : anchorIndex + 1;
129 
130  float startAngle = bezierPath.GetAnchorNormalAngle (anchorIndex) + bezierPath.GlobalNormalsAngle;
131  float endAngle = bezierPath.GetAnchorNormalAngle (nextAnchorIndex) + bezierPath.GlobalNormalsAngle;
132  float deltaAngle = Mathf.DeltaAngle (startAngle, endAngle);
133 
134  int startVertIndex = pathSplitData.anchorVertexMap[anchorIndex];
135  int endVertIndex = pathSplitData.anchorVertexMap[anchorIndex + 1];
136 
137  int num = endVertIndex - startVertIndex;
138  if (anchorIndex == pathSplitData.anchorVertexMap.Count - 2) {
139  num += 1;
140  }
141  for (int i = 0; i < num; i++) {
142  int vertIndex = startVertIndex + i;
143  float t = i / (num - 1f);
144  float angle = startAngle + deltaAngle * t;
145  Quaternion rot = Quaternion.AngleAxis (angle, localTangents[vertIndex]);
146  localNormals[vertIndex] = (rot * localNormals[vertIndex]) * ((bezierPath.FlipNormals) ? -1 : 1);
147  }
148  }

253  TimeOnPathData CalculatePercentOnPathData (float t, EndOfPathInstruction endOfPathInstruction) {
254  // Constrain t based on the end of path instruction
255  switch (endOfPathInstruction) {
256  case EndOfPathInstruction.Loop:
257  // If t is negative, make it the equivalent value between 0 and 1
258  if (t < 0) {
259  t += Mathf.CeilToInt (Mathf.Abs (t));
260  }
261  t %= 1;
262  break;
263  case EndOfPathInstruction.Reverse:
264  t = Mathf.PingPong (t, 1);
265  break;
266  case EndOfPathInstruction.Stop:
267  t = Mathf.Clamp01 (t);
268  break;
269  }
270 
271  int prevIndex = 0;
272  int nextIndex = NumPoints - 1;
273  int i = Mathf.RoundToInt (t * (NumPoints - 1)); // starting guess
274 
275  // Starts by looking at middle vertex and determines if t lies to the left or to the right of that vertex.
276  // Continues dividing in half until closest surrounding vertices have been found.
277  while (true) {
278  // t lies to left
279  if (t <= times[i]) {
280  nextIndex = i;
281  }
282  // t lies to right
283  else {
284  prevIndex = i;
285  }
286  i = (nextIndex + prevIndex) / 2;
287 
288  if (nextIndex - prevIndex <= 1) {
289  break;
290  }
291  }
292 
293  float abPercent = Mathf.InverseLerp (times[prevIndex], times[nextIndex], t);
294  return new TimeOnPathData (prevIndex, nextIndex, abPercent);
295  }

298  public TimeOnPathData CalculateClosestPointOnPathData (Vector3 worldPoint) {
299  float minSqrDst = float.MaxValue;
300  Vector3 closestPoint = Vector3.zero;
301  int closestSegmentIndexA = 0;
302  int closestSegmentIndexB = 0;
303 
304  for (int i = 0; i < localPoints.Length; i++) {
305  int nextI = i + 1;
306  if (nextI >= localPoints.Length) {
307  if (isClosedLoop) {
308  nextI %= localPoints.Length;
309  } else {
310  break;
311  }
312  }
313 
314  Vector3 closestPointOnSegment = MathUtility.ClosestPointOnLineSegment (worldPoint, GetPoint (i), GetPoint (nextI));
315  float sqrDst = (worldPoint - closestPointOnSegment).sqrMagnitude;
316  if (sqrDst < minSqrDst) {
317  minSqrDst = sqrDst;
318  closestPoint = closestPointOnSegment;
319  closestSegmentIndexA = i;
320  closestSegmentIndexB = nextI;
321  }
322 
323  }

325  float t = (closestPoint - GetPoint (closestSegmentIndexA)).magnitude / closestSegmentLength;
326  return new TimeOnPathData (closestSegmentIndexA, closestSegmentIndexB, t);
