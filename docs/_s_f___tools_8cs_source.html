378  public static float DistanceToLine(Vector2 a, Vector2 b, Vector2 point){
379  // Return minimum distance between line segment vw and point p
380  float l2 = Vector2.SqrMagnitude(a - b); // i.e. |w-v|^2 - avoid a sqrt
381  if (l2 == 0.0)
382  return Vector2.Distance(point, a); // v == w case
383  // Consider the line extending the segment, parameterized as v + t (w - v).
384  // We find projection of point p onto the line.
385  // It falls where t = [(p-v) . (w-v)] / |w-v|^2
386  float t = Vector2.Dot(point - a, b - a) / l2;
387  if (t < 0.0)
388  return Vector2.Distance(point, a); // Beyond the 'v' end of the segment
389  else if (t > 1.0)
390  return Vector2.Distance(point, b); // Beyond the 'w' end of the segment
391  Vector2 projection = a + t * (b - a); // Projection falls on the segment
392  return Vector2.Distance(point, projection);
393  }

397  public static bool LineIntersection(Vector2 p0, Vector2 p1, Vector2 p2, Vector2 p3, out Vector2 intersection){
398 
399  intersection = Vector2.zero;
400 
401  Vector2 s1, s2;
402  s1.x = p1.x - p0.x;
403  s1.y = p1.y - p0.y;
404  s2.x = p3.x - p2.x;
405  s2.y = p3.y - p2.y;
406 
407  float s, t, d;
408  d = -s2.x * s1.y + s1.x * s2.y;
409 
410  if(d == 0){
411  return false; // Parallel lines, no intersection
412  }
413 
414  Vector2 pDiff = p0 - p2;
415  s = (-s1.y * pDiff.x + s1.x * pDiff.y) / d;
416  t = ( s2.x * pDiff.y - s2.y * pDiff.x) / d;
417 
418  if (s >= 0 && s <= 1 && t >= 0 && t <= 1)
419  {
420  intersection = p0 + (t * s1);
421  return true; // Intersection!
422  }
423 
424  return false; // No intersection
425  }
