364  static void UpdateAsyncBuildOperations()
365  {
366  foreach (var oper in s_BakeOperations)
367  {
368  if (oper.surface == null || oper.bakeOperation == null)
369  continue;
370 
371  if (oper.bakeOperation.isDone)
372  {
373  var surface = oper.surface;
374  var delete = GetNavMeshAssetToDelete(surface);
375  if (delete != null)
376  AssetDatabase.DeleteAsset(AssetDatabase.GetAssetPath(delete));
377 
378  surface.RemoveData();
379  surface.navMeshData = oper.bakeData;
380  if (surface.isActiveAndEnabled)
381  surface.AddData();
382  CreateNavMeshAsset(surface);
383  EditorSceneManager.MarkSceneDirty(surface.gameObject.scene);
384  }
385  }
386  s_BakeOperations.RemoveAll(o => o.bakeOperation == null || o.bakeOperation.isDone);
387  if (s_BakeOperations.Count == 0)
388  EditorApplication.update -= UpdateAsyncBuildOperations;
389  }
