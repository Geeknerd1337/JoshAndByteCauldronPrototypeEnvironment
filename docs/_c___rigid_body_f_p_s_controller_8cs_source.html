297  /*
298  * two-stage check to see if the player is grounded
299  */
300  public bool isGrounded()
301  {
302  //creating and assigning locations for the raycast tests.
303  float rad = transform.GetComponent<Collider>().bounds.extents.y;
304  Vector3 modUp = new Vector3(rad / 2, 0, 0), modDown = new Vector3(-rad / 2, 0, 0), modLeft = new Vector3(0, 0, -rad / 2), modRight = new Vector3(0, 0, rad / 2);
305  if (transform.GetComponent<Rigidbody>().velocity.y < -.1)
306  {
307  return false;
308  }
309  //below is to visualize the raws drawn for raycast test.
310  if (displayDebugInfo)
311  {
312  UnityEngine.Debug.DrawRay(transform.localPosition, -transform.up, Color.green, groundedCheckdist);
313  UnityEngine.Debug.DrawRay(transform.localPosition + modLeft, -transform.up, Color.green, groundedCheckdist);
314  UnityEngine.Debug.DrawRay(transform.localPosition + modUp, -transform.up, Color.green, groundedCheckdist);
315  UnityEngine.Debug.DrawRay(transform.localPosition + modDown, -transform.up, Color.green, groundedCheckdist);
316  UnityEngine.Debug.DrawRay(transform.localPosition + modRight, -transform.up, Color.green, groundedCheckdist);
317  }
318  //checks if any of the player is close enough to the ground
319  if (Physics.Raycast(transform.localPosition, -transform.up, groundedCheckdist) || Physics.Raycast(transform.localPosition + modLeft, -transform.up, groundedCheckdist) ||
320  Physics.Raycast(transform.localPosition + modUp, -transform.up, groundedCheckdist) || Physics.Raycast(transform.localPosition + modDown, -transform.up, groundedCheckdist) ||
321  Physics.Raycast(transform.localPosition + modRight, -transform.up, groundedCheckdist)) //Many tests ensures a result that feels good to the player.
322  {
323  return (true);
324  }
325  else
326  {
327  return (false);
328  }

332  * handles the camera's Y and the parent's X (therefore also handling the camera's X)
333  */
334  private void MoveCamera(Vector2 rotation)
335  {
336  //does playerobject rotation
337  currentRotation.y += rotation.x * sensitivity;
338  currentRotation.y = Mathf.Repeat(currentRotation.y, 360);
339  transform.rotation = Quaternion.Euler(0, currentRotation.y, 0);
340 
341  //manipulates the child camera to sync with the y value of the parent.
342  cameraRot.x -= rotation.y * sensitivity;
343 
344  //clamps the y rotation of the camera from -maxYangle to maxYangle to prevent the camera inverting itself
345  cameraRot.x = Mathf.Clamp(cameraRot.x, -maxYAngle, maxYAngle);
346 
347  cameraObject.transform.localRotation = Quaternion.Euler(cameraRot.x, 0, 0);
348  }
